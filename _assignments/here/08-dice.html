	void rollTwoDiceNTimes (int);

	int main () {
		int n;
		srand(4984618);

		printf("How many times do you want to roll two dice? > ");
		scanf("%d", &n);

		rollTwoDiceNTimes(n);

	}

	void rollTwoDiceNTimes (int n) {
		int i, a, b;
		int results[12] = {0};

		for (i = 0; i < n; i++) {
			a = rand() % 6 + 1;
			b = rand() % 6 + 1;

			results[a+b-1] += 1;
		}

		printf("Loop ran %d times.\n", i);

		printf("\n");
		printf("%8s: %d occurrances\n", "One", results[0]); // We know one will never occur.
		printf("%8s: %d occurrances\n", "Two", results[1]);
		printf("%8s: %d occurrances\n", "Three", results[2]);
		printf("%8s: %d occurrances\n", "Four", results[3]);
		printf("%8s: %d occurrances\n", "Five", results[4]);
		printf("%8s: %d occurrances\n", "Six", results[5]);
		printf("%8s: %d occurrances\n", "Seven", results[6]);
		printf("%8s: %d occurrances\n", "Eight", results[7]);
		printf("%8s: %d occurrances\n", "Nine", results[8]);
		printf("%8s: %d occurrances\n", "Ten", results[9]);
		printf("%8s: %d occurrances\n", "Eleven", results[10]);
		printf("%8s: %d occurrances\n", "Twelve", results[11]);
	}
