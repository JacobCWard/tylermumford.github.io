--- layout: default title: fibonacci --- {% highlight c++ %} // Tyler Mumford, 2014 // Compute the first N fibonacci numbers #include using namespace std; typedef unsigned long long int num; int main () { num input = 0, reached = 1; num current = 1, prev = 0, prevPrev = 0; cout << "The size of the num type is " << sizeof(num) << endl; cout << "Enter the number of Fibonacci numbers to calculate: "; cin >> input; cout << endl; while (reached <= input) { if (current < prev) { cout << "Overflow." << endl; break; } cout << double(current) << "\t"; prevPrev = prev; prev = current; current = prev + prevPrev; reached++; } return 0; } {% endhighlight %}
